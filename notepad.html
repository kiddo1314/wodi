<script>
const game={total:0,current:1,under:0,words:[],roles:[],vote:null};
const wordPool=[["牛奶","酸奶"],["公交","地铁"],["西瓜","哈密瓜"],["微信","QQ"],["眼镜","墨镜"],["钢笔","铅笔"],["饺子","包子"],["手机","电话"],["玫瑰","月季"]];
function show(id){document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));document.getElementById(id).classList.add("active");}
function start(n){game.total=n;game.current=1;const pair=wordPool[Math.floor(Math.random()*wordPool.length)];game.words=pair;game.roles=[];game.under=Math.floor(Math.random()*n);for(let i=0;i<n;i++)game.roles.push(i===game.under?"under":"normal");showWord();show("page-word");}
function showWord(){const i=game.current-1;const r=game.roles[i];document.getElementById("role").innerText=r==="under"?"你是：卧底":"你是：平民";document.getElementById("word").innerText=r==="under"?game.words[1]:game.words[0];}
function next(){game.current++;if(game.current>game.total){show("page-vote");renderVote();}else showWord();}
function renderVote(){const el=document.getElementById("voteList");el.innerHTML="";for(let i=1;i<=game.total;i++){const d=document.createElement("div");d.className="player";d.innerText=i+"号";d.onclick=()=>{document.querySelectorAll(".player").forEach(x=>x.classList.remove("voted"));d.classList.add("voted");game.vote=i;};el.appendChild(d);}}
function doVote(){if(!game.vote)return alert("请选择投票对象");const win=game.vote-1===game.under;document.getElementById("result").innerText=win?"✅ 平民胜利！":"❌ 卧底胜利！";document.getElementById("info").innerText="平民："+game.words[0]+" ｜ 卧底："+game.words[1];show("page-result");}
function home(){show("page-home");}
</script>
